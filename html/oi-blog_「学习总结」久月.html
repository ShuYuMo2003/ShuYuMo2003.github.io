void grt(int o, int f)
{
    si[o] = 1;

    {
        int ex = edge[i].node;

            continue;
        grt(ex, o);
        si[o] += si[ex];
        MAX = max(si[ex], MAX);
    }
    MAX = max(MAX, sum - si[o]);
    MX[o] = MAX;
    if (MAX < MX[rt])
        rt = o;
}

void solve(int o)
{
    judge[0] = true;
    for (int i = head[o]; i; i = edge[i].nxt)
    {
        int ex = edge[i].node;
        if (vis[ex])
            continue;
        cnt = 0;
        GetAllDis(ex, 0, edge[i].w);
        for (int j = 1; j <= cnt; j++)
        {
            for (int k = 1; k <= m; k++)
            {
                if (qer[k] >= tmp[j])
                    ans[k] |= judge[qer[k] - tmp[j]];
            }
        }
        for (int j = 1; j <= cnt; j++)
        {
            judge[tmp[j]] = 1;
            Q.push(tmp[j]);
        }
    }
    while (!Q.empty())
    {
        judge[Q.front()] = 0;
        Q.pop();
    }
}
