void Prime(int n) {
    for(int i = 2; i <= n; i++){
        if(!np[i]) prime[++tot] = i, d[i] = i;
        for(int j = 1; j <= tot && (x = prime[j] * i) <= n; j++){
            np[x] = 1; d[x] = prime[j];
            if(i % prime[j] == 0) break;
        }
    }
}
